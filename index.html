#include <iostream>
#include <iomanip>
using namespace std;
int** comb(int** a , int row , int col)
{
  int mid = col/2;
  //clear matrix
  for( int i = 0 ; i < row ; i++)
    for( int j = 0 ; j < col ; j++)
      a[i][j] = 0;
      a[0][mid] = 1; //put 1 in the middle of first row
  //build up Pascal's Triangle matrix
  for( int i = 1 ; i < row ; i++)
  {
    for( int j = 1 ; j < col - 1 ; j++)
      a[i][j] = a[i-1][j-1] + a[i-1][j+1];
  }
  return a;
}
